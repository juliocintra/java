		0  1  2  3  4
	      s[T][E][S][T][E]

				S.charAt(0) --> retorna a letra "T"

		0  1  2  3  4
	     s1[T][E][S][T][E]
				s1.indexOf("E") --> retorna 1

		0  1  2  3  4

				S2.substring(1,4) --> Retorna da posição 1 até a posição 3 "EST (É sempre a segunda opção -1)
